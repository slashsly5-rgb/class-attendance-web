```typescript
// src/app/test-db/page.tsx
import { createClient } from '@/lib/supabase/server'

export default async function TestDatabasePage() {
  const supabase = await createClient()

  // Query classes table
  const { data: classes, error: classesError } = await supabase
    .from('classes')
    .select('*')

  // Query students table
  const { data: students, error: studentsError } = await supabase
    .from('students')
    .select('*')

  // Query attendance_sessions table
  const { data: sessions, error: sessionsError } = await supabase
    .from('attendance_sessions')
    .select('*')

  return (
    <div className="p-8 max-w-4xl mx-auto">
      <h1 className="text-3xl font-bold mb-6">Database Connection Test</h1>

      <div className="space-y-6">
        <section className="border rounded-lg p-4">
          <h2 className="text-xl font-semibold mb-2">Classes Table</h2>
          {classesError ? (
            <p className="text-red-600">Error: {classesError.message}</p>
          ) : (
            <p className="text-green-600">
              ✓ Connected successfully. {classes?.length || 0} classes found.
            </p>
          )}
          {classes && classes.length > 0 && (
            <pre className="mt-2 text-xs bg-gray-100 p-2 rounded overflow-x-auto">
              {JSON.stringify(classes, null, 2)}
            </pre>
          )}
        </section>

        <section className="border rounded-lg p-4">
          <h2 className="text-xl font-semibold mb-2">Students Table</h2>
          {studentsError ? (
            <p className="text-red-600">Error: {studentsError.message}</p>
          ) : (
            <p className="text-green-600">
              ✓ Connected successfully. {students?.length || 0} students found.
            </p>
          )}
        </section>

        <section className="border rounded-lg p-4">
          <h2 className="text-xl font-semibold mb-2">Attendance Sessions Table</h2>
          {sessionsError ? (
            <p className="text-red-600">Error: {sessionsError.message}</p>
          ) : (
            <p className="text-green-600">
              ✓ Connected successfully. {sessions?.length || 0} sessions found.
            </p>
          )}
        </section>

        <section className="border rounded-lg p-4 bg-blue-50">
          <h2 className="text-xl font-semibold mb-2">Environment Variables</h2>
          <p className="text-sm text-gray-700">
            Supabase URL: {process.env.NEXT_PUBLIC_SUPABASE_URL ? '✓ Set' : '✗ Missing'}
          </p>
          <p className="text-sm text-gray-700">
            Supabase Anon Key: {process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY ? '✓ Set' : '✗ Missing'}
          </p>
        </section>

        <section className="border rounded-lg p-4 bg-yellow-50">
          <h2 className="text-xl font-semibold mb-2">Expected Result</h2>
          <p className="text-sm">
            All three tables should show "Connected successfully" with 0 records (tables are empty).
            If you see errors, check:
          </p>
          <ul className="list-disc list-inside text-sm mt-2 space-y-1">
            <li>Environment variables are set correctly in Vercel</li>
            <li>Database migration was deployed successfully</li>
            <li>Row Level Security policies allow SELECT operations</li>
            <li>Supabase project is active and healthy</li>
          </ul>
        </section>
      </div>
    </div>
  )
}
```
